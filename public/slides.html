  18. $(".db-list a").click(function() {
  link = $(this);
  $.ajax({url: $(this).attr("href"),
    success: function(src) {
      link.parents("dt").after("<dd>" + src + "</dd>");
      link.unbind('click').click(function() {
        $(this).parents("dt").next("dd").toggle();
        return false;
      })
    }
  });
  return false;
});
